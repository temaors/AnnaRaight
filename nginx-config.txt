server {
    listen 80;
    server_name annaraight.com www.annaraight.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name annaraight.com www.annaraight.com;

    ssl_certificate /etc/letsencrypt/live/annaraight.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/annaraight.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    root /var/www/funnel-app/public;

    # Static files from .next/static - MOST IMPORTANT FOR CSS
    location /_next/static/ {
        alias /var/www/funnel-app/.next/static/;
        expires 365d;
        add_header Cache-Control "public, immutable";
        access_log off;
        # Ensure CSS files have correct MIME type
        location ~* \.css$ {
            add_header Content-Type text/css;
        }
    }

    # Public files including favicon and images
    location ~* \.(jpg|jpeg|png|gif|ico|svg|webp|woff|woff2|ttf|eot)$ {
        root /var/www/funnel-app/public;
        expires 30d;
        add_header Cache-Control "public";
        access_log off;
        try_files $uri @nextjs;
    }

    # JS and CSS files specifically
    location ~* \.(js|css)$ {
        expires 30d;
        add_header Cache-Control "public";
        access_log off;
        try_files $uri @nextjs;
    }

    # Proxy everything else to Next.js
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_buffering off;
    }

    location @nextjs {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_buffering off;
    }
}
